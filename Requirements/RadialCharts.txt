@section Scripts {
    <script src="https://cdn.jsdelivr.net/npm/apexcharts"></script>
    <script>
        function createRadialChart(selector, value, color) {
            var options = {
                chart: {
                    height: 180,
                    type: "radialBar",
                },
                series: [value],
                plotOptions: {
                    radialBar: {
                        hollow: {
                            size: "70%",
                        },
                        track: {
                            background: "#f2f2f2"
                        },
                        dataLabels: {
                            show: false
                        }
                    }
                },
                colors: [color],
            };

            var chart = new ApexCharts(document.querySelector(selector), options);
            chart.render();
        }

        // Chartları çağır
        createRadialChart("#chart3", 50, "#20c997"); // Orders
        createRadialChart("#chart4", 25, "#6f42c1"); // Unique Visitors
        createRadialChart("#chart5", 45, "#d63384"); // Impressions
        createRadialChart("#chart6", 55, "#ffc107"); // Followers
    </script>
}
